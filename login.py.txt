from django.shortcuts import render


def login(request):
    """
        This method is used to display the request for login  online.

        :param request:its a HttpResponse from user.
        :type request: HttpResponse
        :return: this method returns the html page before log in the page to log in website because without log in they can not register themselces.
        :rtype: HttpResponse.
        """
    return render(request,'login.html')
